public class Character {
    private Weapon weapon;

    public void setWeapon(Weapon weapon) {
        this.weapon = weapon;
    }

    public void attack() {
        if (weapon == null) {
            System.out.println("공격할 무기가 없습니다...");
        }else {
            weapon.attack();
        }
    }
}

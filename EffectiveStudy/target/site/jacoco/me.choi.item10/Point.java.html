public class Point {
    private final int x;
    private final int y;

    public Point(int x, int y) {
        this.x = x;
        this.y = y;
    }

    @Override
    public boolean equals(Object obj) {
        Class<? extends Point> aClass = getClass();

        if (obj.getClass() == null || obj.getClass() != getClass())
            return  false;

        Point point = (Point) obj;

        return point.x == x && point.y == y;
    }
}

public class Circle extends Shape{

    private int x;
    private int y;
    private int r;


    public Circle(int x, int y, int r) {
        this.x = x;
        this.y = y;
        this.r = r;
    }

    @Override
    public String toString() {
        return "Circle{" +
                "x=" + x +
                ", y=" + y +
                ", r=" + r +
                '}';
    }

    public Circle copy() throws CloneNotSupportedException {

        Circle clone = (Circle) clone();
        clone.x = x + 1;
        clone.y = y + 1;


        return clone;
    }
}

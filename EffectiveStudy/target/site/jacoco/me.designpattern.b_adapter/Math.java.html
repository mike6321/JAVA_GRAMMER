public class Math {

    public  static double twoTime(double num) {
        return num*2;
    }
    /*Half*/
    public static double half(double num) {
        return num/2;
    }

    /*추가된 알고리즘*/
    public static Double doubled(Double d) {
        return d*2;
    }
